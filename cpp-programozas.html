A kerekasztal-tárgyalások asztala üvegezésre szorul. Írjon programot, amely kiszámítja tetszőleges üveggyűrű elkészítésének árát. A program kérje be a belső és külső sugár mértékét, valamint az üveg árát Ft/m2-ben. Írja ki, hogy mennyibe kerül az üvegezés. 
                        
#include <iostream>

#include <windows.h>

    SetConsoleOutputCP(65001);

    double belsoSugar, kulsoSugar, uvegAr;

    cin >> belsoSugar;

    cout << "Add meg a külső sugár méretét (méterben): ";
    cin >> kulsoSugar;

    cout << "Add meg az üveg árát (Ft/m2-ben): ";
    cin >> uvegAr;

    // Üveggyűrű területének kiszámítása
    terulet = M_PI * (pow(kulsoSugar, 2) - pow(belsoSugar, 2));

    // Üvegköltség kiszámítása
    uvegKoltseg = terulet * uvegAr;

    cout << "Az üvegezés költsége: " << uvegKoltseg << " Ft" << endl;

    system("pause");
    return 0;
}

2.2. feladat

Egy tetszőleges méretű hasáb alakú tartályt töltünk meg vízzel. A tartály méreteit és a liter/másodperc vízhozamot a program kérje be. Számítsa, majd írja ki, hogy mennyi idő alatt telik meg a tartály. 
                        
#include <iostream>
#include <windows.h>

using namespace std;

int main() {
    SetConsoleOutputCP(65001);

    double tartalyMagassag, tartalySzelesseg, tartalyHossz;
    double vizhozam;

    cout << "Add meg a tartály magasságát (méterben): ";
    cin >> tartalyMagassag;

    cout << "Add meg a tartály szélességét (méterben): ";
    cin >> tartalySzelesseg;

    cout << "Add meg a tartály hosszát (méterben): ";
    cin >> tartalyHossz;

    cout << "Add meg a vízhozamot (liter/másodperc): ";
    cin >> vizhozam;

    // Tartály térfogatának kiszámítása
    double tartalyTerfogat = tartalyMagassag * tartalySzelesseg * tartalyHossz;

    // Idő kiszámítása a tartály megteléséhez
    double ido = (tartalyTerfogat) / (vizhozam); // átváltás literből köbméterre

    cout << "A tartály megtelik " << ido << " másodperc alatt." << endl;

    system("pause");
    return 0;
}

Kérjünk be szám karaktereket, írjuk ki, hogy páros vagy páratlan számot adtunk meg, valamint számláljuk meg, hogy hány páros és hány páratlan számot adtunk meg. A ciklust az EOF-ig tartson. Használjuk a getchar() függvényt. Nem szám karakter megadásakor adjunk hibajelzést. 
                        
#include <iostream>
#include <windows.h>

    char karakter;
    int parosDb = 0;
    int paratlanDb = 0;

    while ((karakter = getchar()) != EOF || karakter == '\n') {
        if (karakter >= '0' && karakter <= '9') {
            int szam = karakter - '0';
            if (szam % 2 == 0) {
                cout << "Páros" << endl;
                parosDb++;
            } else {
                cout << "Páratlan" << endl;
                paratlanDb++;
            }
        } else if (karakter != '\n') {
            cout << "Hibás karakter! Kérjük, adjon meg érvényes szám karaktert." << endl;
            cin.ignore();
        }
    }

    cout << "Összesen " << parosDb << " páros és " << paratlanDb << " páratlan számot adott meg." << endl;

    system("pause");
    return 0;
}

#include <iostream>
#include <windows.h>
#include <cmath>

    cout << "Adja meg a 'b' együtthatót: ";
    cin >> b;

    cout << "Adja meg a 'c' együtthatót: ";
    cin >> c;

    double discriminant = pow(b, 2) - 4 * a * c;

    if (discriminant > 0) {
        double x1 = (-b + sqrt(discriminant)) / (2 * a);
        double x2 = (-b - sqrt(discriminant)) / (2 * a);
        cout << "Az egyenlet valós gyökei: x1 = " << x1 << ", x2 = " << x2 << endl;
    } else if (discriminant == 0) {
        double x = -b / (2 * a);
        cout << "Az egyenlet valós gyökei: x1 = x2 = " << x << endl;
    } else {
        cout << "Az egyenletnek nincsenek valós gyökei." << endl;
    }

    system("pause");

}

3.17. feladat

Határozza meg két szám legnagyobb közös osztóját. Használja a korábban ismertetett euklideszi algoritmust. 
                        
#include <iostream>
#include <windows.h>

using namespace std;

int main() {
    SetConsoleOutputCP(65001);

    int szam1, szam2;

    cout << "Adja meg az első számot: ";
    cin >> szam1;

    cout << "Adja meg a második számot: ";
    cin >> szam2;

    /*
    Példa: 84 és 18 legnagyobb közös osztója
    ----------------------------------------
    Először elosztjuk 84-et 18-cal, a hányados 4, a maradék 12.
    Utána elosztjuk 18-at 12-vel, a hányados 1, a maradék 6.
    Ezután elosztjuk 12-t 6-tal a hányados 2, a maradék 0.
    Itt megállt az algoritmus, nincs következő lépés, mivel 0-val nem lehet osztani.
    Tehát az utolsó nem nulla maradék a 6, azaz 84 és 18 legnagyobb közös osztója 6.
    */
    while (szam2 != 0) {
        int temp = szam2;
        szam2 = szam1 % szam2;
        szam1 = temp;
    }

    int legnagyobbKozosOszto = szam1;

    cout << "A két szám legnagyobb közös osztója: " << legnagyobbKozosOszto << endl;

    system("pause");
    return 0;
}
