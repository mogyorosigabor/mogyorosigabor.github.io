    cin >> belsoSugar;

    cout << "Add meg a külső sugár méretét (méterben): ";
    cin >> kulsoSugar;

    cout << "Add meg az üveg árát (Ft/m2-ben): ";
    cin >> uvegAr;

    // Üveggyűrű területének kiszámítása
    terulet = M_PI * (pow(kulsoSugar, 2) - pow(belsoSugar, 2));

    // Üvegköltség kiszámítása
    uvegKoltseg = terulet * uvegAr;

    cout << "Az üvegezés költsége: " << uvegKoltseg << " Ft" << endl;

    system("pause");
    return 0;

2.2. feladat

Egy tetszőleges méretű hasáb alakú tartályt töltünk meg vízzel. A tartály méreteit és a liter/másodperc vízhozamot a program kérje be. Számítsa, majd írja ki, hogy mennyi idő alatt telik meg a tartály. 
                        
#include <iostream>
#include <windows.h>

using namespace std;

int main() {
    SetConsoleOutputCP(65001);

    double tartalyMagassag, tartalySzelesseg, tartalyHossz;
    double vizhozam;

    cout << "Add meg a tartály magasságát (méterben): ";
    cin >> tartalyMagassag;

    cout << "Add meg a tartály szélességét (méterben): ";
    cin >> tartalySzelesseg;

    cout << "Add meg a tartály hosszát (méterben): ";
    cin >> tartalyHossz;

    cout << "Add meg a vízhozamot (liter/másodperc): ";
    cin >> vizhozam;

    // Tartály térfogatának kiszámítása
    double tartalyTerfogat = tartalyMagassag * tartalySzelesseg * tartalyHossz;

    // Idő kiszámítása a tartály megteléséhez
    double ido = (tartalyTerfogat) / (vizhozam); // átváltás literből köbméterre

    cout << "A tartály megtelik " << ido << " másodperc alatt." << endl;

    system("pause");
    return 0;
}

using namespace std;

int main() {
  SetConsoleOutputCP(65001);

  char c; // karaktert tároló változó
  cout << "Karaktereket be SPACE-ig:" << endl; // kiírjuk az utasítást
  do {
    cin.get(c); // beolvassuk a karaktert
    if (!isspace((int)c)) {
      cout << c << " " << (int)c << endl; // kiírjuk a karaktert és az ASCII kódját
    }

#include <windows.h>

    int paratlanDb = 0;

    while ((karakter = getchar()) != EOF || karakter == '\n') {
        if (karakter >= '0' && karakter <= '9') {
            int szam = karakter - '0';
            if (szam % 2 == 0) {
                cout << "Páros" << endl;
                parosDb++;

                cout << "Páratlan" << endl;
                paratlanDb++;
            }

            cout << "Hibás karakter! Kérjük, adjon meg érvényes szám karaktert." << endl;

        }
    }

    return 0;
}

    double discriminant = pow(b, 2) - 4 * a * c;

        double x2 = (-b - sqrt(discriminant)) / (2 * a);

        double x = -b / (2 * a);

3.17. feladat

Határozza meg két szám legnagyobb közös osztóját. Használja a korábban ismertetett euklideszi algoritmust. 
                        
#include <iostream>
#include <windows.h>

using namespace std;

int main() {
    SetConsoleOutputCP(65001);

    int szam1, szam2;

    cout << "Adja meg az első számot: ";
    cin >> szam1;

    cout << "Adja meg a második számot: ";
    cin >> szam2;

    /*
    Példa: 84 és 18 legnagyobb közös osztója
    ----------------------------------------
    Először elosztjuk 84-et 18-cal, a hányados 4, a maradék 12.
    Utána elosztjuk 18-at 12-vel, a hányados 1, a maradék 6.
    Ezután elosztjuk 12-t 6-tal a hányados 2, a maradék 0.
    Itt megállt az algoritmus, nincs következő lépés, mivel 0-val nem lehet osztani.
    Tehát az utolsó nem nulla maradék a 6, azaz 84 és 18 legnagyobb közös osztója 6.
    */
    while (szam2 != 0) {
        int temp = szam2;
        szam2 = szam1 % szam2;
        szam1 = temp;
    }

    int legnagyobbKozosOszto = szam1;

    cout << "A két szám legnagyobb közös osztója: " << legnagyobbKozosOszto << endl;

    system("pause");
    return 0;
}

8.6. feladat

Készítsen egy programot e-mail címek nyilvántartására, az adatok: név és e-mail, ezeket egy struktúra-tömbben tárolja. A tömböt tetszőleges számú elememmel töltse fel. A feltöltéshez és a kiíráshoz használjon függvényt.
Az előző programot egészítse ki olyan függvénnyel, amelyik név szerint keres a tömbben.
Az előző feladatokat egészítse ki olyan függvénnyel, amely a megtalált nevet és a hozzá tartozó címet kitörli a tömbből. 
                        
#include <iostream>
#include <cstring>
#include <windows.h>

using namespace std;

const int MAXN = 100;

int main() {
    SetConsoleOutputCP(65001);

    char nevek[MAXN][100];
    int telefonok[MAXN];

    // Adatok feltöltése
    int i = 0;
    while (i < MAXN) {
        cout << "Adja meg a(z) " << i + 1 << ". nevet (vagy adja meg a # karaktert a befejezéshez): ";
        if (!cin.getline(nevek[i], 100)) {
            break; // Adatbevitel leállítása az EOF jelzésre (Ctrl+Z vagy Ctrl+D)
        }

        if (nevek[i][0] == '#') {
            break;
        }

        cout << "Adja meg a(z) " << i + 1 << ". telefonszámot: ";
        cin >> telefonok[i];

        cin.ignore(); // Buffer ürítése az új sor karakter miatt

        i++;
    }

    cin.clear(); // Az EOF jelzés törlése a bemeneti állapotban

    // Telefonszám bekérése
    int bekertTelefon;
    cout << "Adjon meg egy telefonszámot: ";
    cin >> bekertTelefon;

    // Név keresése és kiírása
    bool talalt = false;
    for (int j = 0; j < i; j++) {
        if (telefonok[j] == bekertTelefon) {
            cout << "A(z) " << bekertTelefon << " telefonszámhoz tartozó név: " << nevek[j] << endl;
            talalt = true;
            break;
        }
    }

    if (!talalt) {
        cout << "Nem található név a(z) " << bekertTelefon << " telefonszámmal." << endl;
    }

    system("pause");
    return 0;
}
